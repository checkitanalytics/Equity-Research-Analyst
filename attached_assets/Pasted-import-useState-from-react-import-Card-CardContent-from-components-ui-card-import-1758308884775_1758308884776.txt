import { useState } from "react";
import { Card, CardContent } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { MessageSquare } from "lucide-react";

const steps = [
  { id: 1, title: "Industry", desc: "Select sector & initial query" },
  { id: 2, title: "Hot Picks", desc: "Search trending stocks" },
  { id: 3, title: "DCF", desc: "Run valuation model" },
  { id: 4, title: "Earnings", desc: "Analyze call transcripts" },
  { id: 5, title: "Sentiment", desc: "Check news & market tone" },
  { id: 6, title: "Observations", desc: "Summarize insights" },
];

export default function EquityResearchPage() {
  const [activeStep, setActiveStep] = useState(1);

  const prompts = {
    1: "Which industry are you interested in exploring today?",
    2: "Here are some hot picks. Do you want me to compare their fundamentals?",
    3: "Shall I run a DCF model for Apple, Tesla, or another stock?",
    4: "Would you like a breakdown of the latest earnings call highlights?",
    5: "Do you want me to summarize sentiment from the latest news?",
    6: "Here are my valuation observations. Want me to suggest an action?",
  };

  const progressPercent = ((activeStep - 1) / (steps.length - 1)) * 100;

  return (
    <div className="flex flex-col h-screen bg-gradient-to-br from-blue-50 to-purple-50">
      {/* Header */}
      <header className="px-6 py-4 shadow bg-white">
        <h1 className="text-2xl font-bold text-indigo-700">
          Checkit Analytics
        </h1>
        <p className="text-sm text-gray-500">AI-Powered Equity Research Analyst</p>
      </header>

      {/* Horizontal Stepper */}
      <div className="bg-white shadow-sm px-10 py-6">
        <div className="flex items-center justify-between">
          {steps.map((step) => (
            <div
              key={step.id}
              className={`flex flex-col items-center cursor-pointer ${
                activeStep === step.id ? "text-indigo-600" : "text-gray-400"
              }`}
              onClick={() => setActiveStep(step.id)}
            >
              <div
                className={`flex items-center justify-center w-10 h-10 rounded-full border-2 ${
                  activeStep === step.id
                    ? "bg-indigo-600 border-indigo-600 text-white"
                    : "border-gray-300"
                }`}
              >
                {step.id}
              </div>
              <span className="mt-2 text-xs font-medium">{step.title}</span>
            </div>
          ))}
        </div>

        {/* Progress Bar */}
        <div className="relative w-full h-2 bg-gray-200 rounded-full mt-6">
          <div
            className="absolute h-2 bg-indigo-600 rounded-full transition-all duration-500"
            style={{ width: `${progressPercent}%` }}
          />
        </div>
      </div>

      {/* Main Content */}
      <div className="flex flex-1 p-6 gap-6">
        {/* Left: Step Details */}
        <Card className="flex-1">
          <CardContent className="p-6">
            <h2 className="text-xl font-semibold text-indigo-700 mb-2">
              Step {activeStep}: {steps.find((s) => s.id === activeStep).title}
            </h2>
            <p className="text-gray-600">
              {steps.find((s) => s.id === activeStep).desc}
            </p>
          </CardContent>
        </Card>

        {/* Right: Investment Assistant */}
        <Card className="flex-1 flex flex-col">
          <CardContent className="p-6 flex flex-col flex-grow">
            <div className="flex items-center mb-4">
              <MessageSquare className="w-5 h-5 text-indigo-600 mr-2" />
              <h2 className="font-semibold text-indigo-700">
                Investment Assistant
              </h2>
            </div>
            <div className="flex-1 bg-gray-50 rounded-lg p-4 mb-4 overflow-auto">
              <p className="text-gray-700">{prompts[activeStep]}</p>
            </div>
            <div className="flex">
              <input
                type="text"
                placeholder="Ask me about investment opportunities..."
                className="flex-1 border border-gray-300 rounded-l-lg p-2 focus:outline-none focus:ring-2 focus:ring-indigo-500"
              />
              <Button className="rounded-r-lg bg-indigo-600 hover:bg-indigo-700">
                Send
              </Button>
            </div>
          </CardContent>
        </Card>
      </div>
    </div>
  );
}
